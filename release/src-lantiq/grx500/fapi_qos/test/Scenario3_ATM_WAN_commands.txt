# replace nas0 with appropriate WAN interface for ATM WAN

cd /opt/lantiq/usr/sbin

qcfg --queue -A q2 -l 100 -i nas0 -m wfq --priority 2 --weight 2 --map 2

classcfg -A cl3 -i br0 --dmac 00:1A:BC:40:EF:80 -j mark --mark 2
